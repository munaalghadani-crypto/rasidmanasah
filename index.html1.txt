    <script>
        // Google Sheet CSV URL
        const SPREADSHEET_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQPoBCYkSfBqOwa27GJ3Dt0KIzXl2DIRqsHkSOODQ83uZKnF5CDAP933kLHLPQvBishBBXVvmcSPW38/pub?output=csv';

        async function fetchAndSearch() {
            const inputId = document.getElementById('civilId').value.trim();
            const errorMsg = document.getElementById('error-msg');
            const btnText = document.getElementById('btnText');
            const btnLoader = document.getElementById('btnLoader');
            const searchBtn = document.getElementById('searchBtn');

            if (!inputId) {
                showError("يرجى إدخال الرقم المدني");
                return;
            }

            errorMsg.classList.add('hidden');
            btnText.innerText = "جاري البحث...";
            btnLoader.classList.remove('hidden');
            searchBtn.disabled = true;

            try {
                const response = await fetch(SPREADSHEET_URL);
                if (!response.ok) throw new Error("Connection failed");
                
                const csvData = await response.text();
                const rows = parseCSV(csvData);
                const studentRow = rows.find(row => row[0] === inputId);

                if (studentRow) {
                    const student = {
                        name: studentRow[1] || "طالبة متميزة",
                        grades: [
                            { label: "درجة الحفظ", score: studentRow[2] || 0, max: 10, emoji: "📖" },
                            { label: "درجة التلاوة", score: studentRow[3] || 0, max: 10, emoji: "🎤" },
                            { label: "السؤال القصير الأول", score: studentRow[4] || 0, max: 5, emoji: "📝" },
                            { label: "السؤال القصير الثاني", score: studentRow[5] || 0, max: 5, emoji: "✍️" },
                            { label: "الواجب المنزلي", score: studentRow[6] || 0, max: 10, emoji: "🏠" },
                            { label: "الاختبار القصير الأول", score: studentRow[7] || 0, max: 10, emoji: "⏱️" },
                            { label: "الاختبار القصير الثاني", score: studentRow[8] || 0, max: 10, emoji: "🎯" }
                        ]
                    };
                    displayResults(student, inputId);
                } else {
                    showError("الرقم المدني غير مسجل في سجلات مدرسة قباء.");
                }
            } catch (error) {
                showError("خطأ في الاتصال بالبيانات. يرجى المحاولة لاحقاً.");
            } finally {
                btnText.innerText = "استعلام عن النتائج";
                btnLoader.classList.add('hidden');
                searchBtn.disabled = false;
            }
        }

        function parseCSV(text) {
            return text.split('\n').map(line => line.split(',').map(cell => cell.replace(/^"(.*)"$/, '$1').trim()));
        }

        function showError(msg) {
            const errorMsg = document.getElementById('error-msg');
            errorMsg.innerText = msg;
            errorMsg.classList.remove('hidden');
        }

        function displayResults(student, id) {
            document.getElementById('display-id').innerText = id;
            document.getElementById('student-name').innerText = student.name;
            const container = document.getElementById('grades-container');
            container.innerHTML = '';

            student.grades.forEach((grade, index) => {
                const percentage = (parseFloat(grade.score) / grade.max) * 100;
                let statusEmoji = "⭐";
                let textColor = "text-emerald-700";
                if (percentage >= 90) { statusEmoji = "🏆"; textColor = "text-amber-700"; }
                else if (percentage >= 75) { statusEmoji = "👏"; textColor = "text-blue-700"; }

                container.innerHTML += `
                    <div class="bg-white p-5 rounded-2xl border border-purple-50 shadow-sm hover:shadow-md transition-all flex justify-between items-center transform hover:-translate-y-1" style="animation: slideUp 0.4s ease forwards ${index * 0.1}s; opacity: 0;">
                        <div class="flex items-center gap-4">
                            <div class="w-10 h-10 flex items-center justify-center text-xl bg-purple-50 rounded-lg">
                                ${grade.emoji}
                            </div>
                            <div>
                                <h5 class="text-purple-900 font-bold text-sm">${grade.label}</h5>
                                <p class="text-[10px] text-purple-400 font-bold uppercase tracking-tight">Max Score: ${grade.max}</p>
                            </div>
                        </div>
                        <div class="flex items-center gap-2">
                            <span class="text-xl font-black ${textColor}">${grade.score}</span>
                            <span>${statusEmoji}</span>
                        </div>
                    </div>
                `;
            });

            document.getElementById('home-screen').classList.add('hidden');
            document.getElementById('student-screen').classList.remove('hidden');
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function goBack() {
            document.getElementById('home-screen').classList.remove('hidden');
            document.getElementById('student-screen').classList.add('hidden');
            document.getElementById('civilId').value = '';
        }
    </script>

        @keyframes fadeIn { from { opacity: 0; } to { opacity: 1; } }
